01

Métodos básicos de instancia

Añadir elementos

add
ArrayList<String> nombres = new ArrayList<>();
nombres.add("Juanito");      // añade al final
nombres.add("Pedro");
nombres.add(5, "Nicolas");  // inserta en [5]

// addAll — añadir otra colección
ArrayList<String> extra = new ArrayList<>();
extra.add("Bilbao"); extra.add("Zaragoza");
nombres.addAll(extra);        // al final
nombres.addAll(1, extra);     // en posición 1

Acceder y modificar

get / set
ArrayList<Integer> nums = new ArrayList<>();
nums.add(5); nums.add(1); nums.add(3);

// get — acceder por índice
int primero = nums.get(0);  // → 5
int tercero = nums.get(2);  // → 3

// set — reemplazar por posición
nums.set(2, "Luis");  // reemplaza [2] por Luis

// size — número de elementos
int tam = nums.size();       // → 3

Eliminar elementos

remove
ArrayList<String> lista = new ArrayList<>();
lista.add("Manuel"); lista.add("Juanito");
lista.add("Pedro");  lista.add("Manuel");

// Por índice
lista.remove(0);          // elimina [0]

// Por valor (1ª ocurrencia)
lista.remove("Manuel");   // solo la primera

// removeAll — diferencia de conjuntos
lista.removeAll(Arrays.asList("Pedro", "Ana"));

// clear — vaciar todo
lista.clear();

Buscar e inspeccionar

contains / indexOf
ArrayList<String> rep = new ArrayList<>();
rep.add("Juanito"); rep.add("Pedro");
rep.add("Juanito"); rep.add("Maria");

// contains — ¿existe el elemento?
rep.contains("Juanito");    // → true
rep.contains("Luis");       // → false

// indexOf — primera posición
rep.indexOf("Juanito");      // → 0

// lastIndexOf — última posición
rep.lastIndexOf("Juanito");  // → 2

// isEmpty — ¿está vacía?
rep.isEmpty();               // → false

Operaciones de conjuntos

retainAll / removeAll
ArrayList<String> A = new ArrayList<>(
  Arrays.asList("Madrid","Barcelona","Valencia","Sevilla")
);
ArrayList<String> B = new ArrayList<>(
  Arrays.asList("Barcelona","Sevilla","Bilbao")
);

// Intersección: mantiene solo comunes
A.retainAll(B);  // → [Barcelona, Sevilla]

// Diferencia: elimina los de B de A
A.removeAll(B);  // → [Madrid, Valencia]

Clonar e inmutables

clone / List.of
// Clonación (copia superficial)
ArrayList<String> original = new ArrayList<>(
  Arrays.asList("Madrid", "Barcelona")
);
ArrayList<String> clon =
  (ArrayList<String>) original.clone();

// Lista inmutable (Java 9+)
List<String> fija = List.of("Manuel", "Pedro");

// Convertir a mutable
List<String> mutable = new ArrayList<>(fija);
mutable.add("Ana"); // ✅ permitido ahora
02

Métodos de java.util.Collections

Ordenamiento

sort / reverse
ArrayList<Integer> nums =
  new ArrayList<>(Arrays.asList(5,3,1,4,2));

// Orden ascendente
Collections.sort(nums);
// → [1, 2, 3, 4, 5]

// Orden descendente
Collections.sort(nums, Collections.reverseOrder());
// → [5, 4, 3, 2, 1]

// Invertir orden actual
Collections.reverse(nums);
// → [1, 2, 3, 4, 5]

Búsqueda y barajeo

binarySearch / shuffle
// binarySearch (lista DEBE estar ordenada)
ArrayList<Integer> ord =
  new ArrayList<>(Arrays.asList(1,2,3,4,5));
Collections.binarySearch(ord, 3); // → 2 (índice)
Collections.binarySearch(ord, 6); // → negativo (no existe)

// shuffle — barajeo aleatorio
Collections.shuffle(ord);

// shuffle con semilla (reproducible)
Collections.shuffle(ord, new Random(42));

Utilidades de lista

fill / copy / nCopies
// fill — llenar con un valor
ArrayList<String> lista =
  new ArrayList<>(Arrays.asList("1","2","3"));
Collections.fill(lista, "X"); // → [X, X, X]

// copy (dest debe tener mismo tamaño)
Collections.copy(destino, origen);

// nCopies — lista inmutable con copias
List<String> x5 =
  Collections.nCopies(5, "Hola");
// → [Hola, Hola, Hola, Hola, Hola]

Estadísticas y análisis

max / min / frequency
ArrayList<Integer> n =
  new ArrayList<>(Arrays.asList(5,31,4,2));

Collections.max(n); // → 31
Collections.min(n); // → 2

// frequency — contar ocurrencias
ArrayList<String> f =
  new ArrayList<>(Arrays.asList("A","B","A","C","A"));
Collections.frequency(f, "A"); // → 3

// disjoint — ¿no tienen elementos comunes?
Collections.disjoint(lista1, lista2); // → true/false
