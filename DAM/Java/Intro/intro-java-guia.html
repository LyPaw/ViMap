Casting (conversión de tipos)

casting
// Implícito (widening) — automático
int i = 42;
double d = i;     // int → double ✅
long l = i;       // int → long ✅

// Explícito (narrowing) — puede perder datos
double pi = 3.99;
int n = (int) pi;  // → 3 (trunca)

long grande = 1234567890L;
int chico = (int) grande;  // ⚠ posible pérdida

// char ↔ int
char c = 'A';
int ascii = c;    // → 65
char c2 = (char) 66; // → 'B'
⚠ Orden wideningbyte → short → int → long → float → double

Operadores

Aritméticos · relacionales · lógicos · bitwise · ternario

Aritméticos

+ − * / %
int a = 10, b = 3;
a + b  // → 13
a - b  // → 7
a * b  // → 30
a / b  // → 3  (entera)
a % b  // → 1  (módulo/resto)

// División real
(double) a / b  // → 3.333…

// Incremento / decremento
a++;   // post: usa luego incrementa
++a;   // pre: incrementa luego usa
a--;   b--;

// Asignación compuesta
a += 5; a -= 2;
a *= 3; a /= 2; a %= 4;

Relacionales y lógicos

== != > < && ||
// Relacionales → boolean
5 == 5   // → true
5 != 3   // → true
7 > 3    // → true
4 <= 4   // → true

// Lógicos
true && false  // → false (AND)
true || false  // → true  (OR)
!true          // → false (NOT)

// Ternario — condición ? si : no
int max = (a > b) ? a : b;
String msg = (edad >= 18)
    ? "mayor" : "menor";

// instanceof
obj instanceof String // → true/false

if / else if / else

condicional
int nota = 75;

if (nota >= 90) {
    System.out.println("Sobresaliente");
} else if (nota >= 70) {
    System.out.println("Notable");
} else if (nota >= 50) {
    System.out.println("Aprobado");
} else {
    System.out.println("Suspenso");
}

switch clásico y expresión

switch
// Switch clásico
switch (dia) {
    case 1: System.out.println("Lunes"); break;
    case 2: System.out.println("Martes"); break;
    default: System.out.println("Otro");
}

// Switch expresión (Java 14+)
String nombre = switch (dia) {
    case 1 -> "Lunes";
    case 2 -> "Martes";
    case 6, 7 -> "Fin de semana";
    default -> "Otro";
};

for — bucles

iteración
// For estándar
for (int i = 0; i < 5; i++) {
    System.out.println(i);
}

// For-each (enhanced for)
int[] nums = {1, 2, 3, 4, 5};
for (int n : nums) {
    System.out.println(n);
}

// Bucle con índice inverso
for (int i = nums.length - 1; i >= 0; i--) {
    System.out.print(nums[i] + " ");
}

while · do-while · break · continue

bucles
// while — condición al inicio
int i = 0;
while (i < 5) {
    System.out.println(i);
    i++;
}

// do-while — ejecuta al menos 1 vez
do {
    System.out.println("al menos una vez");
    i++;
} while (i < 3);

// break — sale del bucle
// continue — salta a la siguiente iteración
for (int j = 0; j < 10; j++) {
    if (j == 5) break;
    if (j % 2 == 0) continue;
    System.out.print(j + " "); // 1 3
}

Métodos

Definición · parámetros · retorno · sobrecarga · recursividad

Estructura de un método

definición
// visibilidad retorno nombre(parámetros)
public static int suma(int a, int b) {
    return a + b;
}

// void — no devuelve nada
public static void saludar(String nombre) {
    System.out.println("Hola, " + nombre);
}

// Varargs — número variable de argumentos
public static int sumarTodos(int... nums) {
    int total = 0;
    for (int n : nums) total += n;
    return total;
}
sumarTodos(1, 2, 3, 4); // → 10

Sobrecarga y paso de parámetros

overloading
// Sobrecarga — mismo nombre, distintos parámetros
public static double area(double radio) {
    return Math.PI * radio * radio;
}
public static double area(double base, double h) {
    return base * h / 2;
}

// Paso por VALOR (primitivos) — copia
void doble(int n) { n *= 2; } // NO modifica original

// Paso por REFERENCIA (objetos) — apunta al mismo
void renombrar(Persona p) {
    p.setNombre("Ana"); // SÍ modifica el objeto
}
RecursividadUn método que se llama a sí mismo. Necesita un caso base para terminar. Ej: factorial(n) = n * factorial(n-1), base: factorial(0) = 1

Programación Orientada a Objetos

Clases · objetos · constructores · encapsulación · this

Anatomía de una clase

clase completa
public class Persona {

  // Atributos (variables de instancia)
  private String nombre;
  private int edad;
  private static int contador = 0;

  // Constructor por defecto
  public Persona() {
      contador++;
  }

  // Constructor con parámetros
  public Persona(String nombre, int edad) {
      this.nombre = nombre;
      this.edad = edad;
      contador++;
  }

  // Constructor encadenado con this()
  public Persona(String nombre) {
      this(nombre, 0);
  }
}
  // Getters (accesores)
  public String getNombre() {
      return nombre;
  }
  public int getEdad() {
      return edad;
  }

  // Setters (mutadores)
  public void setNombre(String nombre) {
      this.nombre = nombre;
  }

  // Método estático
  public static int getContador() {
      return contador;
  }

  // toString
  @Override
  public String toString() {
      return nombre + " (" + edad + ")";
  }

Crear y usar objetos

instancias
// Crear objetos con new
Persona p1 = new Persona("Ana", 25);
Persona p2 = new Persona("Carlos");
Persona p3 = new Persona();

// Usar métodos
p1.getNombre();   // → "Ana"
p1.setNombre("Andrea");
System.out.println(p1); // llama toString()

// Miembro estático — con nombre de clase
Persona.getContador(); // → 3

// null — referencia vacía

// pNull.getNombre(); → NullPointerException ❌

Herencia y Polimorfismo

extends · super · @Override · abstract · final

extends y super

herencia
public class Animal {
    protected String nombre;

    public Animal(String nombre) {
        this.nombre = nombre;
    }
    public void hablar() {
        System.out.println("...");
    }
}

public class Perro extends Animal {
    public Perro(String nombre) {
        super(nombre); // llama al padre
    }
    @Override
    public void hablar() {
        System.out.println(nombre + ": ¡Guau!");
    }
}

Clases abstractas

abstract
// No se puede instanciar directamente
public abstract class Figura {
    protected String color;

    // Método abstracto — sin implementación
    public abstract double area();

    // Método concreto — con implementación
    public void mostrarColor() {
        System.out.println("Color: " + color);
    }
}

public class Circulo extends Figura {
    private double radio;
    @Override
    public double area() {
        return Math.PI * radio * radio;
    }
}

Polimorfismo

polimorfismo
// Variable de tipo padre → objeto hijo
Animal a = new Perro("Rex");
a.hablar(); // → "Rex: ¡Guau!" (del hijo)

// Array polimórfico
Animal[] animales = {
    new Perro("Rex"),
    new Gato("Misu"),
    new Pájaro("Pico")
};
for (Animal animal : animales) {
    animal.hablar(); // cada uno habla a su manera
}

// Casting de objetos
if (a instanceof Perro p) {
    p.buscarPalo(); // Java 16+ pattern matching
}

Interfaces y Clases internas

interface · implements · default · Enum · Record

Interface

contrato
// Interface — contrato de métodos
public interface Volador {
    void volar(); // abstracto por defecto

    // Método default (Java 8+)
    default void aterrizar() {
        System.out.println("Aterrizando...");
    }
}

// Una clase puede implementar VARIAS interfaces
public class Avion extends Vehiculo
                   implements Volador, Transportable {
    @Override
    public void volar() {
        System.out.println("¡Despegando!");
    }
}
interface vs abstractInterface: solo contrato, múltiple. Abstract: puede tener estado, solo una herencia.

Enum · Record

tipos especiales
// Enum — constantes tipadas
public enum DiaSemana {
    LUNES, MARTES, MIERCOLES,
    JUEVES, VIERNES, SABADO, DOMINGO;

    public boolean esFinDeSemana() {
        return this == SABADO || this == DOMINGO;
    }
}
DiaSemana d = DiaSemana.LUNES;
d.esFinDeSemana(); // → false
d.name();          // → "LUNES"
d.ordinal();       // → 0

// Record (Java 16+) — clase de datos inmutable
public record Punto(double x, double y) {}
Punto p = new Punto(3.0, 4.0);
p.x(); p.y(); // getters automáticos

Excepciones

try · catch · finally · throw · throws · checked vs unchecked

try / catch / finally

manejo
try {
    int resultado = 10 / 0;  // lanza excepción
} catch (ArithmeticException e) {
    System.out.println("División por cero: " + e.getMessage());
} catch (NullPointerException e) {
    System.out.println("Null: " + e.getMessage());
} finally {
    // Siempre se ejecuta, haya o no excepción
    System.out.println("fin del bloque");
}

// Multi-catch (Java 7+)
catch (IOException | SQLException e) { ... }

// try-with-resources (cierre automático)
try (FileReader fr = new FileReader("f.txt")) {
    // fr se cierra automáticamente
}

throw · throws · excepciones propias

lanzar
// throw — lanzar una excepción
public void setEdad(int edad) {
    if (edad < 0)
        throw new IllegalArgumentException(
            "Edad no puede ser negativa"
        );
    this.edad = edad;
}

// throws — declara excepciones comprobadas
public void leerArchivo(String ruta)
        throws IOException {
    // puede lanzar IOException
}

// Excepción personalizada
public class SaldoInsuficienteException
        extends Exception {
    public SaldoInsuficienteException(String msg) {
        super(msg);
    }
}
